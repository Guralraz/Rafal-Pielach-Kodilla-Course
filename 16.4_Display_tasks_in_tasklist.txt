SELECT IL.NAME, I.SUMMARY, I.DESCRIPTION
FROM ISSUESLISTS IL JOIN ISSUES I ON IL.ID = I.ISSUESLIST_ID;

SELECT CONCAT(U.FIRSTNAME, ' ', U.LASTNAME) AS USER, I.SUMMARY, I.DESCRIPTION
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.ID;

SELECT CONCAT(U.FIRSTNAME, ' ', U.LASTNAME) AS USER, COUNT(*) AS NUMBER_OF_TASKS
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
